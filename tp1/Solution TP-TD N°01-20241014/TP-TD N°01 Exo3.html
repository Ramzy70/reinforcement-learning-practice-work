def epsilon_greedy(K,epsilons, steps, experiments):
    # Initialize arrays to store results
    average_rewards = np.zeros((len(epsilons), steps))
    optimal_actions = np.zeros((len(epsilons), steps))

    for index, epsilon in enumerate(epsilons):  # for each epsilon value 
        for _ in range(experiments):  # run multiple experiments 

            #Initialization 
            action_values = np.zeros(K) #  q(a)
            action_counts=np.zeros(K) # number of times an action were selected 
            true_action_values = np.random.normal(0, 1, K)  
            optimal_action = np.argmax(true_action_values)


            for step in range(1,steps):
                #update epsilon value
                if index==3:
                    epsilon=1/step
                elif index==4:
                    epsilon=1/step**2 
                
                # select an action using Epsilon-greedy policy 
                if np.random.rand() < epsilon:
                    action = np.random.choice(K)  # select an action randomly Exploration
                else:
                    action = np.argmax(action_values)  # Exploitation
                
                # compute the reward we recieve by performing the selected action 
                reward = np.random.normal(true_action_values[action], 1) 

                # Update q(a) using sample-average method
                action_counts[action] += 1
                action_values[action] += (reward - action_values[action]) / action_counts[action]

                # Calculate the percentage of optimal action
                optimal_actions[index, step] += (action == optimal_action) 
                average_rewards[index, step] += reward 
    return average_rewards/experiments, optimal_actions/experiments

def UCB(K,c_values,num_steps,num_experiments):
    average_rewards = np.zeros((len(c_values), num_steps))
    optimal_actions = np.zeros((len(c_values), num_steps))

    for index, c in enumerate(c_values):  # for each c value 
        for _ in range(num_experiments):  # run multiple experiments 

            #Initialization 
            action_values = np.zeros(K) 
            action_counts=np.ones(K) #  to avoid zero divide error
            true_action_values = np.random.normal(0, 2, K) 
            optimal_action = np.argmax(true_action_values)


            for step in range(1,num_steps+1): # step start from to 1 instead of 0 to void ln(0)
                # If N_t (a) = 0, then a is considered to be a maximizing action
                zero_count_actions = np.where(action_counts == 0)[0]
                if len(zero_count_actions) > 0:
                    # Randomly select from actions with zero counts
                    action = np.random.choice(zero_count_actions)
                else:
                    # Use UCB to select action
                    action = np.argmax(action_values + c * np.sqrt(math.log(step) / action_counts))
                
                
                # compute the reward we recieve by performing the selected action 
                reward = np.random.normal(true_action_values[action], 1/2) 

                # Update q(a) using sample-average method
                action_counts[action] += 1
                action_values[action] += (reward - action_values[action]) / action_counts[action]

                optimal_actions[index, step-1] += (action == optimal_action) 
                average_rewards[index, step-1] += reward 

    return average_rewards/num_experiments, optimal_actions/num_experiments

Interpretation of the results Â¶

Îµ-greedy methodÂ¶

  • $A_t= \begin{cases} \arg\max_a Q(a) & \text{with probability } 1 - \varepsilon \\ \text{a random action} & \text{with probability } \varepsilon \end{cases}$

Greedy methodÂ¶

  • $ \large A_t = \arg\max_a Q_t(a)$

UCB methodÂ¶

  • $\large A_t = \arg\max_a \left(Q_t(a) + c\sqrt{\frac{\log t}{N_t(a)}}\right)$
  • c> 0 controls the degree of exploration.
    • higher value of c --> exploration
    • lower value of c --> exploitaion
  • $N_t(a)$ denotes the number of times that action a has been selected prior to time t
    • If $N_t(a)=0$, then a is considered to be a maximizing action.
  • Square-root term is a measure of the uncertainty or variance in the estimate of aâs value.
