rewards = np.zeros(steps) # Array to store the reward at each time step 

## initialization
action_count=np.zeros(k) # Initialize the number of times each action has been selected
action_values = np.zeros(k) # Initialize the Estimated action-value function Q(a) for each action 

true_action_values = np.random.normal(0, 1, k) # True action values Q*(a) for each action
   

for step in range(steps): 
   
    # Select an action using the epsilon-greedy policy:
    if np.random.rand() < epsilon:
        action = np.random.choice(k)   #With probability epsilon, select a random action (explore), 
    else:
        action = np.argmax(action_values) # with probability (1 - epsilon), select the greedy action

    # Compute the reward
    reward = np.random.normal(true_action_values[action], 1) 


    # Update the estimated action-value function Q(a) using the sample-average method
    action_count[action] += 1 
    action_values[action] += (reward - action_values[action]) / action_count[action]

    # Save the reward for the current step
    rewards[step]+=reward

## Hyperparametrs 
k=10  # Number of arms (actions)
steps=2000 # Number of time steps per experiment
epsilon=0.1 # Exploration probability
nb_experiment=1000 # Number of experiments 

## tracking 
average_rewards = np.zeros(steps) # Array to store the average reward at each time step 
optimal_actions = np.zeros(steps) # Array to store the percentage of optimal action selections at each time step


## Apply the algorithms
for _ in range(nb_experiment):
    action_count=np.zeros(k) # Initialize the number of times each action has been selected
    action_values = np.zeros(k) # Initialize the Estimated action-value function Q(a) for each action
    true_action_values = np.random.normal(0, 1, k) # True action values Q*(a) for each action
    optimal_action = np.argmax(true_action_values)  # The optimal action is the one with the highest true action value Q*(a)
    for step in range(steps): 
       
        # Select an action using the epsilon-greedy policy:
        if np.random.rand() < epsilon:
            action = np.random.choice(k)   #With probability epsilon, select a random action (explore), 
        else:
            action = np.argmax(action_values) # with probability (1 - epsilon), select the greedy action

        # Compute the reward
        reward = np.random.normal(true_action_values[action], 1) 


        # Update the estimated action-value function Q(a) using the sample-average method
        action_count[action] += 1 
        action_values[action] += (reward - action_values[action]) / action_count[action]

        # Update the average reward at the current step
        average_rewards[step]+=reward/nb_experiment

        # Update the percentage of selecting the optimal action at the current step
        optimal_actions[step] += (action == optimal_action) / nb_experiment

            action_count[action] += 1

            average_rewards[index,step]+=reward/nb_experiment

Result Interpretation Â¶

  • When we run a single experiment, we get a fluctuating reward plot due to high variance in the rewards.

  • To smooth out (reduce the fluctuations), we can average the results over multiple experiments.

  • Greedy method (Îµ=0)

    • It improved slightly faster than the other methods at the very beginning but then plateaued at a lower level.
    • It performed significantly worse in the long run because it often got stuck performing suboptimal actions.
    • It found the optimal action in only approximately one-third of the tasks.

  • Îµ-greedy methods performed better than the greedy method because they continued to explore, improving their chances of finding the optimal action.

    • The Îµ = 0.1 method explored more and usually found the optimal action earlier.
    • The Îµ = 0.01 method improved more slowly, but eventually performed better than the Îµ = 0.1 method in long run.

  • Policy with Îµ=0.01 performs best in the long run, because this policy will select the optimal action 10 times more frequently than the policy with Îµ=0.1
