def render_policy(policy, maze, start_state,exit_state):
    nrows, ncols=maze.shape
    _, ax2 =plt.subplots()
    ax2.set_xticks(np.arange(0.5, nrows, step=1))
    ax2.set_xticklabels([])
    ax2.set_yticks(np.arange(0.5, ncols, step=1))
    ax2.set_yticklabels([])
    ax2.grid(True)

    # "bs" -> blue square
    # "gs" -> green square
    ax2.plot(*start_state, "s", markersize=30)  
    ax2.text(*start_state, "start", ha="center", va="center", color="white")

    ax2.plot(*exit_state, "gs", markersize=30)  # exit is a big green square
    ax2.text(*exit_state, "Exit", ha="center", va="center", color="white")

    # render policy
    free_cells = [(row, col)  
            for row in range(nrows) 
                for col in range(ncols) 
                if maze[row][col] == 0
        ]
    
    for cell in free_cells:
        if isinstance(policy[cell], (np.int64, np.int32)) : # check if policy is deterministic or stochastic
            dx = 0
            dy = 0
            #actions = [up, down, left, right] = [0,1,2,3] 
            if policy[cell] == 0:
                dy = -0.2
            if policy[cell] == 1:
                dy = 0.2
            if policy[cell] == 2:
                dx = -0.2
            if policy[cell] == 3:
                dx = 0.2
            ax2.arrow(cell[1],cell[0], dx, dy, color="red", head_width=0.1, head_length=0.1)
        else:
            for (i,value) in enumerate(policy[cell]):
                dx = 0
                dy = 0
                if i == 0:
                    dy = -0.5*value
                if i == 1:
                    dy = 0.5*value
                if i == 2:
                    dx = -0.5*value
                if i == 3:
                    dx = 0.5*value

                # *cell are the starting coordinates of the arrow.
                # dx and dy are the arrow's length and direction.
                ax2.arrow(cell[1],cell[0], dx, dy, color="red", head_width=0.1, head_length=0.1)
                
    ax2.imshow(maze, cmap="binary")
    plt.show()

def getNextState(state,action):
    action=action_space[action]
    nxtState=(state[0]+action[0],state[1]+action[1])
   
    # check if the next state legal
    if (nxtState[0]>=0 and nxtState[0]<nrows and nxtState[1]>=0 and nxtState[1]<ncols):
        if maze[nxtState]==0:

        return state

def step(state,action):
    nxtState=getNextState(state,action)
    reward=get_reward(state,nxtState)
    return nxtState,reward,nxtState==exit_state

def q_learning(num_eps=num_episodes):
    epsl=0.5
    q_table=np.zeros((nrows, ncols,nA))
    for _ in range(num_eps):
        s=start_state
        done=False
        time_step=0
        while not done :
            a=epsilon_greedy(q_table[s],epsl)
            nxs, r, win=step(s,a)
            q_table[s[0],s[1],a]+=alpha*(r+gamma*np.max(q_table[nxs])- q_table[s[0],s[1],a])
            s=nxs
            time_step+=1
            done=win or  time_step==time_limit
    policy=np.argmax(q_table, axis=2)
    return policy

def double_q_learning(num_eps=num_episodes):
    q1_table=np.zeros((nrows, ncols,nA))
    q2_table=np.zeros((nrows, ncols,nA))
    epsl=0.9
    reduction = epsl/num_eps  #*****************
    for _ in range(num_eps):
        s=start_state
        done=False
        time_step=0
        while not done :
            a=epsilon_greedy(q1_table[s]+q2_table[s],epsl)
            nxs, r, win=step(s,a)
            if np.random.rand() <=0.5:
                q1_table[s[0],s[1],a]+=alpha*(r+gamma*(q2_table[nxs][np.argmax(q1_table[nxs])])- q1_table[s[0],s[1],a])
            else:
                q2_table[s[0],s[1],a]+=alpha*(r+gamma*(q1_table[nxs][np.argmax(q2_table[nxs])])- q2_table[s[0],s[1],a])
            s=nxs
            time_step+=1
            done=win or  time_step==time_limit
        if epsl > 0:
            epsl -= reduction
    policy1=np.argmax(q1_table, axis=2)
    policy2=np.argmax(q2_table, axis=2)
    return policy1, policy2

def sarsa(num_eps=num_episodes):
    epsl=epsilon
    reduction = epsl/num_eps  #*****************
    q_table=np.zeros((nrows, ncols,nA))
    for _ in range(num_eps):
        s=start_state
        done=False
        time_step=0
        a=epsilon_greedy(q_table[s],epsl)
        while not done :
            nxs, r, win=step(s,a)
            a2=epsilon_greedy(q_table[nxs],epsl)
            q_table[s[0],s[1],a]+=alpha*(r+gamma*q_table[nxs[0],nxs[1],a2]- q_table[s[0],s[1],a])
            s=nxs
            a=a2
            time_step+=1
            done=win or  time_step==time_limit
        if epsl > 0:
            epsl -= reduction
    policy=np.argmax(q_table, axis=2)
    return policy

What is the main difference between MC, DP and TD learning methods?¶

  • DP requires complete knowledge of the environment, while TD and MC methods do not require a model of the environment (model-free).
  • DP and TD uses bootstrapping to estimate value functions (updating estimates based on other estimates), while MC uses returns from sample episodes, which requires waiting for episode termination.
